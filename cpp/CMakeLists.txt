cmake_minimum_required(VERSION 3.13)
set(CMAKE_VERBOSE_MAKEFILE on)

add_compile_options(
        -fexceptions
        -frtti
        -std=c++20
        -Wall
        -Wpedantic
        -DFOLLY_NO_CONFIG=1
        -DLOG_TAG=\"ReactNative\")

file(GLOB RNMMKV_SOURCES CONFIGURE_DEPENDS *.cpp)
add_library(react_native_mmkv_core STATIC ${RNMMKV_SOURCES})

target_include_directories(react_native_mmkv_core PUBLIC .)

target_link_libraries(react_native_mmkv_core
        fbjni
        jsi
        react_nativemodule_core
)


# For Gradle Syncs (i.e. when you first open the project in Android Studio),
# the react_codegen_RNMmkvSpec target is not existing yet as it's generated by
# Codegen. Therefore skip the linking with that library if it's missing in the
# CMake dependency Graph. It will be included anyway in the final build.
if (TARGET react_codegen_RNMmkvSpec)
    target_include_directories(react_codegen_RNMmkvSpec PUBLIC .)
    target_link_libraries(react_native_mmkv_core react_codegen_RNMmkvSpec)
endif ()
